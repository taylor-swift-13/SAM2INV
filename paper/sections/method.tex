%% ============================================================
\section{Method}\label{sec:method}
%% ============================================================

%% ---- Notation ----
\paragraph{Notation.}
A program $P = (\phi_{\mathrm{pre}}, G, B, \phi_{\mathrm{post}})$
has variables $V$, precondition $\phi_{\mathrm{pre}}$, guard $G$,
body $B$, and postcondition $\phi_{\mathrm{post}}$.
An execution trace $\tau = (\sigma_0, \dots, \sigma_T)$ records
per-iteration states $\sigma_t : V \to \mathbb{Z}$.
A candidate invariant $I = c_1 \wedge \cdots \wedge c_k$ is
\emph{inductive} (written $\mathrm{Ind}(I,P)$) if
$\phi_{\mathrm{pre}} \Rightarrow I$,\;
$I \wedge G \wedge B \Rightarrow I'$,\; and
$I \wedge \neg G \Rightarrow \phi_{\mathrm{post}}$.

%% ------------------------------------------------------------
\subsection{Loop Context Construction}\label{sec:loop-context}
%% ------------------------------------------------------------

Before calling the LLM, we assemble a structured \emph{loop context}
that provides all information needed for invariant synthesis.
A lightweight static pre-pass (\texttt{LoopAnalysis}) extracts a
per-loop \emph{record} $\mathbf{r}_i$ containing:
(i)~the loop guard and body code snippet,
(ii)~the set of modified variables $V_{\mathrm{mod}}$ and unmodified
     function parameters $\mathcal{P}$,
(iii)~conservative integer bounds on each variable, and
(iv)~the loop entry pre-condition $\phi_{\mathrm{pre}}$.

A symbolic execution pass then enriches each record with a
\emph{transition relation}: for each variable the pass computes the
one-step mapping $v \mapsto e'(v, \mathcal{P})$ under the loop guard,
distinguishing changed variables from unchanged ones.
This gives the LLM a concise algebraic view of the loop body beyond
raw source code.

The final prompt concatenates four components:
(i)~the ACSL pre-condition $\phi_{\mathrm{pre}}$,
(ii)~the transition relation summary,
(iii)~the loop code snippet, and
(iv)~execution traces with temporal labels (\textsc{Before} iteration
     / \textsc{After} termination) emphasising that invariants must
     hold at the \emph{start} of every iteration.
Available variables and function parameters (supporting
\lstinline|\at(var, Pre)| references) are listed explicitly so that
the LLM only uses well-scoped identifiers.
The full prompt template is given in Appendix~\ref{app:prompts}.

%% ------------------------------------------------------------
\subsection{Parallel LLM Generation}\label{sec:generation}
%% ------------------------------------------------------------

We issue $N$ parallel calls to the teacher $\mathcal{M}_T$ with
temperatures drawn from $\{0.8, 1.0, 1.1, 1.2\}$.
Each call $j$ returns $r_j = (\rho_j, I_j)$: a CoT reasoning trace
$\rho_j$ and a candidate invariant $I_j$.
Candidates are decomposed into conjuncts and unioned:
\begin{equation}\label{eq:pool}
C_{\mathrm{pool}} = \bigcup_{j=1}^{N}
  \mathrm{conjuncts}(I_j)
  \;\big/\; {\sim_{\mathrm{ws}}}
\end{equation}
where $\sim_{\mathrm{ws}}$ is whitespace-normalised string equality.
The union exploits the conjunctive structure of invariants: distinct
calls discover complementary conjuncts.
Prompt design and temperature strategy are detailed in
Appendix~\ref{app:prompts}.

%% ------------------------------------------------------------
\subsection{Multi-Stage Filtering}\label{sec:filtering}
%% ------------------------------------------------------------

The pool is refined through five stages of increasing cost:
\[
C_{\mathrm{pool}}
\xrightarrow{F_1} C_1
\xrightarrow{F_2} C_2
\xrightarrow{F_3} C_3
\xrightarrow{F_4} C_4
\xrightarrow{F_5} C^*
\]

Algorithm~\ref{alg:pipeline} presents the complete procedure.

\begin{algorithm}[t]
\caption{\sam{} Pipeline}\label{alg:pipeline}
\rule{\linewidth}{0.4pt}
\begin{algorithmic}
\alginput{Program $P$, teacher $\mathcal{M}_T$, verifier $\mathcal{V}$, quality judge $\mathcal{M}_Q$}
\algoutput{Certified invariant $C^*$ or $\bot$; training records}
\algblank
\algline $\mathcal{T} \leftarrow \textsc{TieredSample}(P)$ \cmnt{Phase I}\\
\algline \kw{for} $j = 1,\dots,N$ \kw{in parallel do}\\
\algline \algind $(\rho_j, I_j) \leftarrow \mathcal{M}_T(P, \mathcal{T}, t_j)$\\
\algline $C_{\mathrm{pool}} \leftarrow \bigcup_j \mathrm{conjuncts}(I_j) \;/\; {\sim_{\mathrm{ws}}}$\\
\algblank
\algline $C_1 \leftarrow \{c \in C_{\mathrm{pool}} \mid \mathrm{SynValid}(c)\}$ \cmnt{$F_1$: syntax}\\
\algline $C_2 \leftarrow \{c \in C_1 \mid \forall\sigma\in\mathcal{T}:\,\sigma\models c\}$ \cmnt{$F_2$: trace}\\
\algline $C_3 \leftarrow \mathrm{SemDedup}(C_2)$ \cmnt{$F_3$: dedup}\\
\algblank
\algline $S \leftarrow C_3$ \cmnt{$F_4$: Houdini}\\
\algline \kw{repeat}\\
\algline \algind $R \leftarrow \{c \in S \mid \mathcal{V}(P,\textstyle\bigwedge S, c) = \mathit{fail}\}$\\
\algline \algind $S \leftarrow S \setminus R$\\
\algline \kw{until} $R = \emptyset$ \kw{or} $S = \emptyset$\\
\algline $C_4 \leftarrow S$\\
\algblank
\algline $C^* \leftarrow C_4$ \kw{if} $\mathcal{M}_Q(P,C_4) = \textsc{ok}$, else $\emptyset$ \cmnt{$F_5$: trivial}\\
\algblank
\algline $\rho^* \leftarrow \mathcal{M}_R(P, C^*)$ \cmnt{reverse CoT}\\
\algline \kw{return} $C^*$,\; $y^+ = (\rho^*, C^*)$,\; $\{y^-\} = C_{\mathrm{pool}} \setminus C^*$
\end{algorithmic}
\rule{\linewidth}{0.4pt}
\end{algorithm}

\paragraph{$F_1$: Syntax filter.}
$C_1 = \{c \in C_{\mathrm{pool}} \mid \mathrm{SynValid}(c)\}$
checks ACSL structural rules (Appendix~\ref{app:syntax-gate}).
Cost: $O(|c|)$ per conjunct.

\paragraph{$F_2$: Trace semantic filter.}
$C_2 = \{c \in C_1 \mid
  \forall\tau\!\in\!\mathcal{T},\,
  \forall\sigma\!\in\!\tau:\,\sigma \models c\}$.
Sound (no true invariant removed) but incomplete.

\paragraph{$F_3$: Semantic deduplication.}
An auxiliary LLM identifies equivalence classes under logical
equivalence $\equiv_{\mathrm{sem}}$ and retains one representative
per class:
$C_3 = \{\mathrm{repr}([c]_{\equiv_{\mathrm{sem}}}) \mid c \in C_2\}$.

\paragraph{$F_4$: Houdini pruning.}
The Houdini algorithm~\citep{flanagan2001houdini} iteratively removes
conjuncts whose verification conditions fail in Frama-C/WP until a
fixed point $C_4$.

\begin{theorem}[Termination]\label{thm:termination}
Houdini terminates in $\le |C_3|$ iterations, removing $\ge 1$
conjunct per round.
\end{theorem}

\paragraph{$F_5$: Trivial invariant filter.}
A quality discriminator $\mathcal{M}_Q$ excludes invariants that are
formally correct but semantically vacuous, preventing reward hacking.

%% ------------------------------------------------------------
\subsection{Reverse Chain-of-Thought}\label{sec:reverse-cot}
%% ------------------------------------------------------------

The certified $C^*$ is assembled from fragments of multiple teacher
responses and differs from any single $I_j$.
A reverse CoT model $\mathcal{M}_R$ synthesises a matching reasoning
chain:
$\rho^* = \mathcal{M}_R(P, C^*)$,
ensuring every training sample is a complete $(\rho, I)$ pair.

%% ------------------------------------------------------------
\subsection{Offline Knowledge Distillation}\label{sec:offline}
%% ------------------------------------------------------------

For each program $P$ the pipeline yields:
\begin{itemize}[leftmargin=1.5em,itemsep=1pt]
\item \textbf{Chosen}: $y^+ = (\rho^*, C^*)$ --- certified invariant
  with reverse CoT.
\item \textbf{Rejected}: $\{y^-_s\}$ --- every candidate eliminated
  at stage $s$.
\end{itemize}

\paragraph{Data augmentation.}
(i)~\emph{Off-policy conjunct recombination}
(Algorithm~\ref{alg:aug-mixed}): the pipeline collects a
\emph{negative pool} $B$ of every conjunct rejected at any stage
(syntax, trace, Houdini).
Rejected variants are constructed by independently sampling a
subset of correct conjuncts and a subset of incorrect ones,
subject to the constraint that the result is not equivalent to
the chosen annotation (Appendix~\ref{app:dpo-aug}).
(ii)~\emph{On-policy negatives}: the student $\mathcal{M}_S$
generates candidates through the same pipeline; failures serve as
targeted rejected samples close to the student's current distribution.

\paragraph{Two-stage training.}
\emph{SFT} minimises negative log-likelihood on chosen samples:
\begin{equation}\label{eq:sft}
\mathcal{L}_{\mathrm{SFT}} =
  -\!\sum_{(P,y^+)}\!
  \log \pi_\theta(y^+ \mid P).
\end{equation}
\emph{DPO}~\citep{rafailov2023direct} then refines preferences from
the SFT checkpoint $\pi_{\mathrm{ref}}$:
\begin{equation}\label{eq:dpo}
\mathcal{L}_{\mathrm{DPO}} =
  -\!\sum_{(P,y^+,y^-)}\!
  \log\sigma\!\Bigl(
    \beta\bigl[
      \log\frac{\pi_\theta(y^+\!\mid\! P)}
               {\pi_{\mathrm{ref}}(y^+\!\mid\! P)}
      - \log\frac{\pi_\theta(y^-\!\mid\! P)}
               {\pi_{\mathrm{ref}}(y^-\!\mid\! P)}
    \bigr]
  \Bigr).
\end{equation}
The pipeline certifies every output regardless of model quality: a
weaker model yields a smaller $C^*$ but never an uncertified one.
Stage-stratified DPO weighting and further details are in
Appendix~\ref{app:dpo}.
